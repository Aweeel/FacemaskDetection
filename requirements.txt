h5py==3.15.1            # loading/saving .h5 model files

# ─────────────────────────────────────────────
#  HOW TO RUN
# ─────────────────────────────────────────────
#
#  1. Install Python 3.11 from https://www.python.org/downloads/
#     (during install tick "Add Python to PATH")
#
#  2. Open a terminal in this folder and install dependencies:
#       pip install -r requirements.txt
#
#  3. Start the web app:
#       python app.py
#
#  4. Open your browser and go to:
#       http://localhost:5000
#
#  5. Click Start → allow camera access → click Detect
#
# ─────────────────────────────────────────────
#  RETRAIN THE MODEL (optional)
# ─────────────────────────────────────────────
#
#  Delete the old model first (only needed when changing architecture):
#       del models\face_obstruction_model2.h5
#
#  Then run:
#       python train_model.py
#
#  Each subsequent run of train_model.py builds on the previous session.
